<!DOCTYPE html>
<html lang="en">
<head>
  <meta charset="UTF-8" />
  <title>Simple MIDI Player</title>
</head>
<body>
  <h1>Simple MIDI Player</h1>
  <button id="playPause">Play</button>

  <div class="controls">
    <label>Tempo: <span id="tempoValue">120</span> BPM</label>
    <input id="tempo" type="range" min="40" max="200" value="120" step="1" />
  </div>

  <!-- Tone.js (Web Audio synth + transport) -->
  <script src="https://unpkg.com/tone"></script>
  <!-- @tonejs/midi (MIDI ‚Üí Tone-friendly data) -->
  <script src="https://unpkg.com/@tonejs/midi"></script>

  <script>
    let part = null;
    let isPlaying = false;
    let midiLoaded = false;
    const playBtn = document.getElementById("playPause");
    const tempoSlider = document.getElementById("tempo");
    const tempoValue = document.getElementById("tempoValue");

    Tone.Transport.bpm.value = Number(tempoSlider.value);

    tempoSlider.addEventListener("input", () => {
      const bpm = Number(tempoSlider.value);
      tempoValue.textContent = bpm;
      Tone.Transport.bpm.value = bpm;
    });

    async function loadMidi() {
      const midi = await Midi.fromUrl("your-song.mid");
      const synth = new Tone.PolySynth(Tone.Synth).toDestination();
      const notes = midi.tracks[0].notes;

      part = new Tone.Part((time, note) => {
        synth.triggerAttackRelease(note.name, note.duration, time, note.velocity);
      }, notes);

      part.loop = false;
      part.start(0);
      midiLoaded = true;
    }

    playBtn.addEventListener("click", async () => {
      await Tone.start();
      if (!midiLoaded) await loadMidi();

      if (!isPlaying) {
        Tone.Transport.start();
        isPlaying = true;
        playBtn.textContent = "Stop";
      } else {
        Tone.Transport.stop();
        Tone.Transport.position = "0:0:0";
        isPlaying = false;
        playBtn.textContent = "Play";
      }
    });
  </script>
</body>
</html>
